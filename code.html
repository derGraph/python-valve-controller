import matplotlib.pyplot as plt
#Anfangswasserstand (m)
wasserstand = 1
#Beckengroeße (m2)
becken = 4
#Anfangszufluss (m3)
zufluss = 1
#Sollhoehe (m)
sollhoehe = 1
#Duesenauslass (m3)
duese = 1
###RECHENVARIABLEN
#Fehlervariable
error = 0
p = 2
i = 4
time = 0
wasserliste = []
plt.ion()
#Animationsschleife bis 20
while time < 20:
    #nach 8 Zeiteinheiten wird Zufluss auf 1.4 erhoeht
    if time == 8:
        zufluss = 1.4

    #berechnung der Duesenstellung
    duese += (wasserstand-sollhoehe)*p - (error-(wasserstand-sollhoehe))*i
    error = wasserstand-sollhoehe

    #Wasserstand in Physikmodell ändern
    wasserstand = wasserstand + (zufluss - duese) / becken
    wasserliste.append(wasserstand)

    #Graph zeichnen
    plt.clf()
    plt.plot(wasserliste)
    plt.title("Zuflussänderung von 1.0 auf 1.4 bei 10")
    plt.ylabel("Wasserhoehe (m)")
    plt.xlabel("Zeit")
    plt.pause(0.0000001)

    #erhoehen der Zeit
    time += 1
#Anzeigen des Endergebnisses
plt.show(block=True)
